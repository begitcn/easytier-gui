private func statusIcon(for status: ConnectionStatus) -> String {
    switch status {
    case .connected: return "checkmark.circle.fill"

    case .disconnected: return "circle"
    case .error: return "exclamationmark.circle.fill"

private func statusColor(for status: ConnectionStatus) -> Color {

    case .connected: return .green
    case .connecting: return .orange
    case .disconnected: return .secondary
    case .error: return .red
    }

private func statusText(for status: ConnectionStatus) -> String {

    case .disconnected: return "未连接"
    case .error: return "错误"
    }
}

```swift
private func statusBadge(for status: ConnectionStatus) -> some View {
    HStack(spacing: 6) {
        Image(systemName: statusIcon(for: status))
            .font(.system(size: 10))
            .foregroundColor(statusColor(for: status))
        
        Text(statusText(for: status))
            .font(.system(size: 11, design: .rounded))
            .foregroundColor(statusColor(for: status))
    }
    .padding(.horizontal, 10)
    .padding(.vertical, 5)
    .background(statusColor(for: status).opacity(0.1))
    .cornerRadius(12)
}

/// 更新菜单栏图标基于连接状态
func updateStatusIcon(isRunning: Bool, hasError: Bool) {
    guard let button = statusItem.button else { return }
    
    let iconName: String
    if hasError {
        iconName = "network.badge.shield.half.filled"  // 错误状态
    } else if isRunning {
        iconName = "network.badge.shield.half.filled"  // 运行中
    } else {
        iconName = "network"  // 未连接
    }
    
    if let image = NSImage(systemSymbolName: iconName, accessibilityDescription: "EasyTier 状态") {
        image.isTemplate = true
        button.image = image
    }
}
```

3. 在 ProcessViewModel 或 NetworkRuntime 状态变化时调用：
```swift
.onChange(of: isRunning) { _, newValue in
    menuBarManager.updateStatusIcon(isRunning: newValue, hasError: false)
}
.onChange(of: hasError) { _, newValue in
    menuBarManager.updateStatusIcon(isRunning: isRunning, hasError: newValue)
}
```

4. 添加颜色变化支持（可选，使用 tinted image）：
```swift
func updateStatusIconColored(isRunning: Bool, hasError: Bool) {
    guard let button = statusItem.button else { return }
    
    let config: String
    let color: NSColor
    
    if hasError {
        config = "network.badge.shield.half.filled"
        color = .systemRed
    } else if isRunning {
        config = "network.badge.shield.half.filled"
        color = .systemGreen
    } else {
        config = "network"
        color = .labelColor
    }
    
    if let image = NSImage(systemSymbolName: config, accessibilityDescription: nil) {
        let coloredImage = image.withSymbolConfiguration(
            .init(paletteColors: [color])
        )
        button.image = coloredImage
    }
}
```
