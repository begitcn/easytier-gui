## Task 2: 按钮悬停效果增强

**Type:** execute
**Files:** EasyTierGUI/Views/ConnectionView.swift

<read_first>
- EasyTierGUI/Views/ConnectionView.swift (搜索 Button 或 .buttonStyle)
</read_first>

<action>
为关键按钮添加悬停效果：

1. 创建自定义按钮样式扩展：
```swift
// MARK: - Hover Button Style

struct HoverButtonStyle: ButtonStyle {
    let normalOpacity: Double
    let hoverOpacity: Double
    
    init(normal: Double = 1.0, hover: Double = 0.85) {
        self.normalOpacity = normal
        self.hoverOpacity = hover
    }
    
    func makeBody(configuration: Configuration) -> some View {
        configuration.label
            .opacity(configuration.isPressed ? 0.7 : (configuration.isHovered ? hoverOpacity : normalOpacity))
            .animation(.quick, value: configuration.isHovered)
            .animation(.quick, value: configuration.isPressed)
    }
}
```

2. 在关键按钮上应用：
```swift
// 连接按钮
Button(action: { vm.connect() }) {
    HStack {
        Image(systemName: "play.fill")
        Text("启动网络")
    }
    .frame(maxWidth: .infinity)
    .padding(.vertical, 12)
    .background(Color.green.opacity(0.15))
    .foregroundColor(.green)
    .cornerRadius(10)
}
.buttonStyle(HoverButtonStyle(normal: 1.0, hover: 0.85))
.disabled(vm.isRunning)

// 断开按钮
Button(action: { vm.disconnect() }) {
    HStack {
        Image(systemName: "stop.fill")
        Text("停止网络")
    }
    .frame(maxWidth: .infinity)
    .padding(.vertical, 12)
    .background(Color.red.opacity(0.15))
    .foregroundColor(.red)
    .cornerRadius(10)
}
.buttonStyle(HoverButtonStyle(normal: 1.0, hover: 0.85))
```

3. 对于图标按钮（工具栏按钮），使用类似效果：
```swift
Button(action: { ... }) {
    Image(systemName: "plus.circle.fill")
        .font(.system(size: 24))
        .foregroundColor(.green)
        .scaleEffect(configuration.isHovered ? 1.1 : 1.0)
        .animation(.quick, value: configuration.isHovered)
}
.buttonStyle(.plain)
```
</action>

<read_first>
- EasyTierGUI/Views/PeersView.swift (节点列表行样式)

<action>
为节点列表行添加悬停效果和选中状态：

1. 在 PeersView 中添加行样式：
```swift
struct PeerRowStyle: ViewModifier {
    @State private var isHovered = false
    @State private var isSelected = false
    
    func body(content: Content) -> some View {
        content
            .background(
                RoundedRectangle(cornerRadius: 8)
                    .fill(isSelected ? Color.accentColor.opacity(0.15) : (isHovered ? Color.primary.opacity(0.05) : Color.clear))
            )
            .animation(.quick, value: isHovered)
            .animation(.quick, value: isSelected)
            .onHover { hovering in
                isHovered = hovering
            }
    }
}
```

2. 应用到节点列表行：
```swift
ForEach(Array(filteredPeers.enumerated()), id: \.element.id) { index, peer in
    PeerRowView(peer: peer, index: index)
        .modifier(PeerRowStyle())
        .id(peer.id)
}
```

3. 或使用 SwiftUI 内置的 .listRowHoverEffect（macOS 14+）：
```swift
ForEach(...) { peer in
    PeerRowView(peer: peer)
        .listRowHoverEffect(.spring())
}
```

在 SettingsView 中应用统一的动画效果：

1. 为设置页面卡片添加入场动画：
```swift
struct SettingsView: View {
    @EnvironmentObject var vm: ProcessViewModel
    @State private var appear = false
    
    var body: some View {
        ScrollView {
            VStack(spacing: CGFloat.cardSpacing) {
                // 设置卡片
                GeneralSettingsCard()
                    .opacity(appear ? 1 : 0)
                    .offset(y: appear ? 0 : 10)
                    .animation(.standard.delay(0.1), value: appear)
                
                CoreUpdateCard()
                    .opacity(appear ? 1 : 0)
                    .offset(y: appear ? 0 : 10)
                    .animation(.standard.delay(0.2), value: appear)
                
                AboutCard()
                    .opacity(appear ? 1 : 0)
                    .offset(y: appear ? 0 : 10)
                    .animation(.standard.delay(0.3), value: appear)
            }
            .padding(CGFloat.spacingXL)
        }
        .frame(maxWidth: .infinity, maxHeight: .infinity)
        .background(Color(NSColor.windowBackgroundColor))
        .onAppear {
            appear = true
        }
    }
}
```

2. 为卡片内容添加交错动画效果（可选）：
```swift
// 在子视图卡片中

}
```
